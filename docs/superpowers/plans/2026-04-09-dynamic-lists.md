# Dynamic Lists Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Compile `.map()` over arrays into pre-allocated piu Label slots that update on scroll, enabling scrollable lists on Pebble Alloy with zero runtime allocation.

**Architecture:** The compiler uses TypeScript AST analysis to detect `.map()` calls in JSX, identifies the data array and visible window size, then emits N pre-allocated Label slots in the piu template. A Behavior holds the data array and a scroll index; `refresh()` updates each slot's `.string` and `.visible` based on the current window position.

**Tech Stack:** TypeScript AST (`ts` module, already in compiler), Preact (mock render), piu (on-watch output).

**Spec:** `docs/superpowers/specs/2026-04-09-dynamic-lists-design.md`

```typescript
// ---------------------------------------------------------------------------
// List (.map()) detection via AST
// ---------------------------------------------------------------------------

interface ListInfo {
  /** Name of the data array variable (e.g., 'ITEMS') */
  dataArrayName: string;
  /** The array literal values, if statically determinable */
  dataArrayValues: string[] | null;
  /** Number of visible slots (from .slice(start, start + N)) */
  visibleCount: number;
  /** Name of the scroll index state setter (to look up slot index) */
  scrollSetterName: string | null;
  /** Number of labels per item in the .map() callback */
  labelsPerItem: number;
}

function detectListPatterns(exName: string): ListInfo | null {
  const sf = parseExampleSource(exName);
  if (!sf) return null;

  let mapCallFound = false;
  let dataArrayName: string | null = null;
  let visibleCount = 3; // default
  let scrollSetterName: string | null = null;
  let labelsPerItem = 1;
  let dataArrayValues: string[] | null = null;

  walkAST(sf, (node) => {
    // Find: someArray.map(callback) in JSX expression
    if (
      ts.isCallExpression(node) &&
      ts.isPropertyAccessExpression(node.expression) &&
      node.expression.name.text === 'map'
    ) {
      mapCallFound = true;
      const obj = node.expression.expression;
      if (ts.isIdentifier(obj)) {
        dataArrayName = obj.text;
      }

      // Count JSX text elements in the callback to determine labelsPerItem
      if (node.arguments.length > 0) {
        const callback = node.arguments[0]!;
        let textCount = 0;
        walkAST(callback, (n) => {
          if (ts.isJsxElement(n) || ts.isJsxSelfClosingElement(n)) {
            const tagName = ts.isJsxElement(n)
              ? n.openingElement.tagName
              : n.tagName;
            if (ts.isIdentifier(tagName) && tagName.text === 'Text') {
              textCount++;
            }
          }
        });
        if (textCount > 0) labelsPerItem = textCount;
      }
    }

    // Find: const visible = ITEMS.slice(index, index + N)
    if (
      ts.isVariableDeclaration(node) &&
      node.initializer &&
      ts.isCallExpression(node.initializer) &&
      ts.isPropertyAccessExpression(node.initializer.expression) &&
      node.initializer.expression.name.text === 'slice'
    ) {
      const sliceArgs = node.initializer.arguments;
      if (sliceArgs.length >= 2) {
        const secondArg = sliceArgs[1]!;
        // Look for: index + N
        if (ts.isBinaryExpression(secondArg) &&
            secondArg.operatorToken.kind === ts.SyntaxKind.PlusToken &&
            ts.isNumericLiteral(secondArg.right)) {
          visibleCount = Number(secondArg.right.text);
        }
        // Look for the scroll index variable name in the first arg
        if (ts.isIdentifier(sliceArgs[0]!)) {
          // Find which useState setter corresponds to this variable
          // by looking at the source for const [indexVar, setterVar] = useState
          walkAST(sf, (n) => {
            if (
              ts.isVariableDeclaration(n) &&
              ts.isArrayBindingPattern(n.name) &&
              n.name.elements.length >= 2 &&
              !ts.isOmittedExpression(n.name.elements[0]!) &&
              ts.isIdentifier(n.name.elements[0]!.name) &&
              n.name.elements[0]!.name.text === (sliceArgs[0] as ts.Identifier).text
            ) {
              const setter = n.name.elements[1]!;
              if (!ts.isOmittedExpression(setter) && ts.isIdentifier(setter.name)) {
                scrollSetterName = setter.name.text;
              }
            }
          });
        }
      }
    }

    // Find: const ITEMS = ['Alpha', 'Beta', ...] — top-level array literal
    if (
      ts.isVariableDeclaration(node) &&
      ts.isIdentifier(node.name) &&
      node.initializer &&
      ts.isArrayLiteralExpression(node.initializer)
    ) {
      const name = node.name.text;
      const values: string[] = [];
      let allStrings = true;
      for (const el of node.initializer.elements) {
        if (ts.isStringLiteral(el)) {
          values.push(el.text);
        } else {
          allStrings = false;
          break;
        }
      }
      if (allStrings && values.length > 0) {
        // Store — we'll match by name later
        if (dataArrayName === null || name === dataArrayName) {
          dataArrayName = name;
          dataArrayValues = values;
        }
      }
    }
  });

  if (!mapCallFound || !dataArrayName) return null;

  return {
    dataArrayName,
    dataArrayValues,
    visibleCount,
    scrollSetterName,
    labelsPerItem,
  };
}
```

- [ ] **Step 1: Name list slot labels during emitNode**

The pebble-dom tree from mock render has the .map() labels at specific positions. We need to identify them and give them `name: "ls{N}"` attributes. The simplest approach: after detecting `listInfo`, record which label indices correspond to list items.

During the baseline render, the .map() produces `visibleCount` labels. These appear as sequential labels in the tree AFTER any header labels. We can identify them by diffing: render with scroll index 0, then with scroll index 1 — the labels that change text are the list slots.

Add this after the existing perturbation pipeline (after `process.stderr.write('Structural branches:...')` ~line 716):

```typescript
// ---------------------------------------------------------------------------
// List slot detection: which labels are .map() items?
// ---------------------------------------------------------------------------

const listSlotLabels = new Set<number>(); // label indices that are list slots
let listScrollSlotIndex = -1;

if (listInfo) {
  // Find which state slot is the scroll index
  if (listInfo.scrollSetterName && setterSlotMap.has(listInfo.scrollSetterName)) {
    listScrollSlotIndex = setterSlotMap.get(listInfo.scrollSetterName)!;
  }

  // Identify list slot labels by perturbing the scroll index
  // The labels whose text changes when scroll index changes are the list items
  if (listScrollSlotIndex >= 0) {
    forcedStateValues.set(listScrollSlotIndex, 1); // scroll to position 1
    resetStateTracking();
    silence();
    const appScroll = exampleMain();
    restore();

    if (appScroll) {
      const ctxScroll: EmitContext = {
        skins: new Map(), styles: new Map(), declarations: [],
        skinIdx: 0, styleIdx: 0, labelIdx: 0, labelTexts: new Map(),
        rectIdx: 0, rectFills: new Map(),
      };
      emitNode(appScroll._root, ctxScroll, '', null, null);

      for (const [idx, baseText] of t1Texts) {
        const scrollText = ctxScroll.labelTexts.get(idx);
        if (scrollText !== undefined && scrollText !== baseText) {
          listSlotLabels.add(idx);
        }
      }
      appScroll.unmount();
    }
    forcedStateValues.delete(listScrollSlotIndex);
  }

  process.stderr.write(`List slot labels: ${[...listSlotLabels].join(', ')}\n`);
}
```
